ALTER  PROCEDURE [dbo].[sp_Olt_BulkInsert_Txnfile]
@Path Varchar(250),
@FormatFilePath Varchar(250),
@NoOfTxns varchar(10)
AS
BEGIN
set	nocount on
Declare @Query Varchar(2500)

Truncate table TempOfflineData

Set @Query = " BULK INSERT TempOfflineData FROM  '" + @Path   + "' WITH  (formatfile ='" + @FormatFilePath + "',FIELDTERMINATOR = '~', ROWTERMINATOR = '\n',LASTROW = " + @NoOfTxns + ",KEEPNULLS) "  
print @Query
Exec(@Query)
End
